# Filter out entries where Medal is NaN (no medal won)
medals_df = df.dropna(subset=['Medal'])
# Group by gender and medal type to count medals per type for each gender
medal_gender = medals_df.groupby(['Sex','Medal']).size().unstack().reindex(['M','F'])
# Creating a bar chart comparing medal counts by gender and type
medal_gender.plot(kind='bar', figsize=(8,6), color=['gold','silver','peru'],edgecolor='black')
plt.title('Medals by Rank and Gender')
plt.xlabel('Gender')
plt.ylabel('Number of Medals')
plt.xticks(rotation=0)
plt.legend(title='Medal type')
plt.tight_layout()
plt.savefig('medals_by_type_and_gender.png', dpi=300, bbox_inches='tight')
plt.show()

# Group the data by year and gender to count athletes per gender per year
gender_counts = df.groupby(['Year','Sex'])['ID'].count().reset_index()
gender_counts_pivot = gender_counts.pivot(index='Year',columns='Sex', values='ID')

# Plotting the number of athletes by gender for each Olympic year
plt.figure(figsize=(10,6))
plt.plot(gender_counts_pivot.index, gender_counts_pivot['M'], label='Male', color = 'blue', linewidth=2)
plt.plot(gender_counts_pivot.index, gender_counts_pivot['F'], label='Female', color = 'red', linewidth=2)

plt.title('Athlete count by Gender over Years')
plt.xlabel('Year')
plt.ylabel('Number of Athletes')
plt.legend(title='Gender')
plt.grid(True)

plt.savefig('Gender_count_by_year.png',dpi=300,bbox_inches='tight')
plt.show()
